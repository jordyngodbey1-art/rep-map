// 1. Link to State Shapes const statesDataUrl = 'https://raw.githubusercontent.com/PublicaMundi/MappingPHL/master/data/us_states.json'; // 2. Load and Process Data Promise.all([ fetch('./reps.json').then(res => res.json()), fetch(statesDataUrl).then(res => res.json()) ]).then(([reps, statesData]) => { // Group reps by state: { "Florida": ["Rep A", "Rep B"], "Texas": ["Rep C"] } const stateGroups = {}; reps.forEach(r => { if (!stateGroups[r.state]) stateGroups[r.state] = []; stateGroups[r.state].push(r.name); }); // 3. Add States to Map L.geoJson(statesData, { style: function(feature) { const count = (stateGroups[feature.properties.name] || []).length; return { fillColor: count > 5 ? '#003366' : count > 2 ? '#004a99' : count > 0 ? '#4da3ff' : '#e0e0e0', weight: 1.5, opacity: 1, color: 'white', fillOpacity: 0.75 }; }, onEachFeature: function(feature, layer) { const stateName = feature.properties.name; const repsInState = stateGroups[stateName] || []; // Create a clean list for the popup let repListHtml = repsInState.length > 0 ? repsInState.map(name => `
  • ${name}
    • `).join('') : "
  • No Rep Assigned
    • "; layer.bindPopup(`

    ${stateName}

    ${repsInState.length} Reps Found:

      ${repListHtml}
    `); // Hover effects layer.on({ mouseover: (e) => { e.target.setStyle({ fillOpacity: 0.9, weight: 3 }); }, mouseout: (e) => { e.target.setStyle({ fillOpacity: 0.75, weight: 1.5 }); } }); } }).addTo(map); });
